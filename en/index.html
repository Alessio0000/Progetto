 // SPDX-License-Identifier: MIT

 pragma solidity ^0.8.17;

  contract Fundraiser2{
         
    uint public totalRaised;         // total amount raised
    uint public goal;                // goal to reach
    uint public totalDonations;      // total amount of donations
    address payable public manager;  // manager's address
    bool public closed;              // variable to check if the fundraiser is open or not
    bool public goalmet;             // variable to check if the goal is met or not
     
       
    // The manager is whoever deploys the contract
    // goal is set on deploying the contract
    constructor(uint _goal) {
        manager = payable (msg.sender);
        goal = _goal;
        goalmet = false;
    }
    fallback() external payable { } 
    receive() external payable { }

    modifier onlyOwner {
        require (msg.sender == manager, "Only the manager can do that");
        _;
    }
    //function that gets donations from whichever address calls it, 
    //updates totalRaised and toatlDonations
    function getDonation() payable public{
        require (goalmet == false, "Fundraiser is completed");
        require (closed == false, "Fundraiser is closed");
        payable (this).transfer(msg.value);
        totalRaised = totalRaised + (msg.value);
        totalDonations ++;
    }
    //function that lets only the manager withdraw the funds that have been raised
    //NOTE: do not withdraw any funds before closing the fundraiser or any eth remaining will be lost
    function withdrawFunds() payable public onlyOwner{
        require(totalRaised >= goal, "Goal is not met yet");
        payable (msg.sender).transfer(address (this).balance);
        closed = true;
    }
    
    //function that lets only the owner close the fundraiser and transfers 
    //any remaining funds to his address
    function closeFundraiser() payable public onlyOwner{
        require(closed == false, "Fundraiser is already closed");
          uint remainingFunds = address (this).balance;
        if (remainingFunds > 0){
            manager.transfer(remainingFunds);
        }
        closed = true;
    }

    //function that checks whether the goal is reached or not
    function checkStatus() public returns (string memory){
        
        if (totalRaised >= goal) {
            goalmet = true;
            return "Congratulations, fundraiser complete";
        }
        else {
            goalmet = false;
            return "Goal is not met yet, fundraiser is still going";
          }
      }
  }
